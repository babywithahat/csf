    \item Problem 12
    \begin{itemize}
      \item a)\\ 
      $[\lnot p\land(p\lor q)]\to q$\\
      $\equiv[(\lnot p\land p)\lor (\lnot p\land q)]\to q$ \hfill Distributive Law\\
      $\equiv [$ F $ \lor (\lnot p\land q)]\to q$ \hfill Negation Law\\
      $\equiv (\lnot p\land q)\to q$ \hfill Identity Law\\ 
      $\equiv \lnot (\lnot p\land q)\lor q$ \hfill Conditional Equivalence\\ 
      $\equiv p\lor \lnot q\lor q$ \hfill De Morgan's Law\\ 
      $\equiv p\lor $ T \hfill Negation Law\\ 
      $\equiv$ T \hfill Domination Law 
      \item b)\\
      $[(p\to q)\land (q\to r)]\to (p\to r)$\\
      $\equiv [(p\lor q)\to r]\to (p\to r)$\hfill Conditional Equivelence\\
      $\equiv \lnot [\lnot(p\lor q)\lor r]\lor (\lnot p\lor r)$\hfill Conditional Equivelences\\
      $\equiv [\lnot (\lnot p\land \lnot q)\land \lnot r]\lor (\lnot p\lor r)$\hfill De Morgan's Law\\
      $\equiv [(p\lor q)\land \lnot r]\lor (\lnot p\lor r)$\hfill De Morgan's Law\\
      $\equiv [(\lnot p\lor r)\lor (p\lor q)]\land [(\lnot p\lor r)\lor \lnot r]$\hfill Distributive Law\\
      $\equiv [$ T $ \lor r\lor q]\land [$ T $ \lor \lnot p]$\hfill Negation Law\\
      $\equiv $ T $ \land $ T \hfill Domination Law\\
      $\equiv $ T 
      \item c)\\
      $[p\land (p\to q)]\to q$\\
      $\equiv \lnot [p\land (\lnot p\lor q)]\lor q$\hfill Conditional Equivelence\\
      $\equiv [\lnot p\lor (p\land \lnot q)]\lor q$\hfill De Morgan's Law\\
      $\equiv (\lnot p\lor p\lor q)\land (\lnot p\lor \lnot q\lor q)$\hfill Distributive Law\\ 
      $\equiv ($ T $ \lor q)\land (\lnot p\lor $ T $)$\hfill Negation Law\\
      $\equiv $ T $\land $ T \hfill Domination Law\\
      $\equiv $ T \hfill Identity Law
      \item d)\\
      $[(p\lor q)\land (p\to r)\land (q\to r)]\to r$\\
      $\equiv \lnot [(p\lor q)\land (\lnot p\lor r)\land (\lnot q\lor r)]\lor r$\hfill Conditional Equivelence\\
      $\equiv (\lnot p\land \lnot q)\lor (p\land \lnot r)\lor (q\land \lnot r)\lor r$\hfill De Morgan's Law\\
      $\equiv [\lnot p\lor (p\land \lnot r)] \land [\lnot q\lor (p\land \lnot r)]\lor (q\land \lnot r)\lor r$\hfill Distr. Law\\
      $\equiv [(\lnot p\lor p)\land(\lnot p\lor \lnot r)] \land [\lnot q\lor (p\land \lnot r)]\lor (q\land \lnot r)\lor r$\hfill Ds. Law\\
      $\equiv (\lnot p\lor \lnot r) \land [\lnot q\lor (p\land \lnot r)]\lor (q\land \lnot r)\lor r$\hfill Neg. \& Id. Law\\
      $\equiv (\lnot p\lor \lnot r) \land [\lnot q\lor (p\land \lnot r)]\lor (q\lor r)\land (\lnot r\lor r)$\hfill Ds. Law\\
      $\equiv (\lnot p\lor \lnot r) \land [\lnot q\lor (p\land \lnot r)]\lor (q\lor r)$\hfill Neg. \& Id. Law\\
